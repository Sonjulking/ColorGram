
CREATE TABLE users (
    user_no NUMBER PRIMARY KEY,
    user_nickname VARCHAR2(100) NOT NULL,
    user_id VARCHAR2(100) NOT NULL UNIQUE,
    user_password VARCHAR2(100) NOT NULL,
    user_email VARCHAR2(100) UNIQUE
);
